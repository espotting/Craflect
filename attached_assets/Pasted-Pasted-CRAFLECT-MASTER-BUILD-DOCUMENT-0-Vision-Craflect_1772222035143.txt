6. Schéma SQL
CREATE TABLE users (id UUID PRIMARY KEY,email TEXT,created_at TIMESTAMP DEFAULT NOW());

CREATE TABLE workspaces (id UUID PRIMARY KEY,owner_id UUID,name TEXT,created_at TIMESTAMP DEFAULT NOW());

CREATE TABLE content_sources (id UUID PRIMARY KEY,workspace_id UUID,type TEXT,title TEXT,file_url TEXT,transcript TEXT,created_at TIMESTAMP DEFAULT NOW());

CREATE TABLE generated_content (id UUID PRIMARY KEY,source_id UUID,format TEXT,hook_type TEXT,text TEXT,video_url TEXT,created_at TIMESTAMP DEFAULT NOW());

CREATE TABLE briefs (id UUID PRIMARY KEY,workspace_id UUID,topic TEXT,hook TEXT,script TEXT,format TEXT,created_at TIMESTAMP DEFAULT NOW());

CREATE TABLE performance (id UUID PRIMARY KEY,generated_content_id UUID,platform TEXT,views INT,engagement FLOAT,retention FLOAT,created_at TIMESTAMP DEFAULT NOW());

CREATE TABLE events (id UUID PRIMARY KEY,user_id UUID,event_name TEXT,metadata JSONB,created_at TIMESTAMP DEFAULT NOW());
